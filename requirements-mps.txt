# Apple Silicon (MPS) packages for hydra-suite
# Install with: uv pip install -v -r requirements-mps.txt
#
# Compatible with: M1, M2, M3, M4 chips (macOS 12.3+)
# Base dependencies are in pyproject.toml (installed via -e . below).
# This file adds torch and MPS-specific packages only.

# PyTorch with MPS support (included in standard PyTorch wheel for macOS ARM64)
torch
torchvision
torchaudio

# ONNX model export and ONNX Runtime inference (CoreMLExecutionProvider is used when available)
onnx
onnxslim
onnxruntime

# Install this package in editable mode (pulls all pyproject.toml dependencies)
-e .
